组合数

组合数（求$C_a^b$模上p的值）

根据数据范围来选择

$C_a^b$ 看a和b的范围

a, b $\le$ 2000

$C_a^b = C_{a-1}^b + C_{a-1}^{b-1}$ 可以通过递推来计算

#include <iostream>
#include <algorithm>

using namespace std;

const int N = 2010, mod = 1e9 + 7;

int c[N][N];

void init() {
    for (int i = 0; i < N; i ++)
        for (int j = 0; j <= i; j ++)
            if (!j) c[i][j] = 1;
            else c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % mod;
}

int main() {
    init();
    
    int n;
    scanf("%d", &n);
    while (n --) {
        int a, b;
        scanf("%d%d", &a, &b);
        printf("%d\n", c[a][b]);
    }
    
    return 0;
}

a, b $\le$ $10^{5}$

预处理$\frac{a!}{(a-b)!b!}$中的阶乘

除数的取余需要用到逆元，预处理

#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;

typedef long long LL;

const int N = 100010, mod = 1e9 + 7;

int n;
int fact[N], infact[N]; // infact表示逆元

int qmi(int a, int k, int mod) {
    int res = 1 % mod;
    while (k) {
        if (k & 1) res = (LL)res * a % mod;
        a = (LL)a * a % mod;
        k >>= 1;
    }
    return res;
}

void init() {
    fact[0] = infact[0] = 1;
    for (int i = 1; i < N; i ++ ) {
        fact[i] = (LL)fact[i - 1] * i % mod;
        infact[i] = (LL)infact[i - 1] * qmi(i, mod - 2, mod) % mod;
    }
}

int main() {
    init();
    
    scanf("%d", &n);
    while (n --) {
        int a, b;
        scanf("%d%d", &a, &b);
        printf("%d\n", (LL)fact[a] * infact[b] % mod * infact[a - b] % mod);
    }
    
    return 0;
}

a, b $\le$ $10^{18}$

mod上p的值，p是质数

数据量大，使用卢卡斯定理：$C_a^b\equiv C_{a, mod, p}^{b,mod,p} * C_{a/p}^{b/p},(mod,p)$

#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;

typedef long long LL;

const int N = 100010;

int n, p;

int qmi(int a, int k) {
    int res = 1 % p;
    while (k) {
        if (k & 1) res = (LL)res * a % p;
        a = (LL)a * a % p;
        k >>= 1;
    }
    return res;
}

int C(int a, int b) {
    int res = 1;
    for (int i = 1, j = a; i <= b; i ++, j --) {
        res = (LL)res * j % p;
        res = (LL)res * qmi(i, p - 2) % p;
    }
    return res;
}

int lucas(LL a, LL b) {
    if (a < p && b < p) return C(a, b);
    return (LL)C(a % p, b % p) * lucas(a / p, b / p) % p;
}

int main() {
    scanf("%d", &n);
    while (n --) {
        LL a, b;
        scanf("%lld%lld%lld", &a, &b, &p);
        printf("%d\n", lucas(a, b));
    }
    return 0;
}

答案不取模时使用高精度

卡特兰数

$C_{2n}^n - C_{2n}^{n-1} = C_{2n}^n / (n + 1)$

给定n个0和n个1，它们将按照某种顺序排成长度为2n的序列，求它们能排列成的所有序列中，能够满足任意前缀序列中0的个数都不少于1的个数的序列有多少个。

输出的答案对10^9^+7取模。

输出格式

共一行，包含整数n。

输出格式

共一行，包含一个整数，表示答案。

数据范围

1≤n≤10^5^

输入样例：

3

输出样例：

5
#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;

typedef long long LL;

const int mod = 1e9 + 7, N = 200010;

int n;

int qmi(int a, int k, int p) {
    int res = 1 % p;
    while (k) {
        if (k & 1) res = (LL)res * a % p;
        a = (LL)a * a % p;
        k >>= 1;
    }
    return res;
}

int main() {
    cin >> n;
    
    int res = 1;
    for (int i = 2 * n; i > n; i --) res = (LL)res * i % mod;
    for (int i = 1; i <= n; i ++) res = (LL)res * qmi(i, mod - 2, mod) % mod;
    
    res = (LL)res * qmi(n + 1, mod - 2, mod) % mod;
    
    cout << res << endl;
    
    return 0;
}
